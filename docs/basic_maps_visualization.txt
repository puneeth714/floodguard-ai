It handles asynchronous Google bootstrap initialization, active hook management, and dynamic state triggers for multi-layered toggles:jsximport React, { useState, useEffect, useRef } from 'react';
import { APIProvider, Map, useMap, Marker } from '@vis.gl/react-google-maps';

// --- CUSTOM WRAPPER FOR NATIVE GOOGLE VISUALIZATION HEATMAPS ---
const HeatmapLayer = ({ points }) => {
  const map = useMap(); // Safely pulls the parent initialized map object instance
  const heatmapRef = useRef(null);

  useEffect(() => {
    // Prevent execution if Google maps engine core instance is not ready yet
    if (!map || !window.google || !window.google.maps.visualization) return;

    // Convert standard JSON data coordinates into native Google engine locations
    const googlePoints = points.map(
      p => new window.google.maps.LatLng(p.lat, p.lng)
    );

    // Initialize map overlay engine parameters
    heatmapRef.current = new window.google.maps.visualization.HeatmapLayer({
      data: googlePoints,
      map: map,
      radius: 30,       // Blur blur radius surrounding nodes
      maxOpacity: 0.8   // Point core structural center opacity limits
    });

    // Cleanup layer immediately on unmount to protect memory stacks
    return () => {
      if (heatmapRef.current) {
        heatmapRef.current.setMap(null);
      }
    };
  }, [map, points]);

  return null; // This rendering component is functional-only (canvas overlay)
};

// --- MAIN MULTI-LAYER MAP CONTROLLER COMPONENT ---
export default function MultiLayerMap() {
  // Layer visibility toggles
  const [showHeatmap, setShowHeatmap] = useState(true);
  const [showMarkers, setShowMarkers] = useState(true);

  // Sample Data Coordinates Set
  const heatmapData = [
    { lat: 37.7749, lng: -122.4194 },
    { lat: 37.7752, lng: -122.4178 },
    { lat: 37.7785, lng: -122.4210 },
    { lat: 37.7820, lng: -122.4110 }
  ];

  const markerData = [
    { id: 'store-1', lat: 37.7891, lng: -122.4014, title: "Flagship Store" },
    { id: 'store-2', lat: 37.7512, lng: -122.4456, title: "Regional Warehouse" }
  ];

  return (
    // Wrap target views inside the global API Context script loader
    <APIProvider 
      apiKey="YOUR_GOOGLE_MAPS_API_KEY" 
      libraries={['visualization']} // MANDATORY flag parameter for heatmaps
    >
      <div style={{ padding: '15px', fontFamily: 'sans-serif' }}>
        
        {/* State Toggle Buttons Layout Control Block */}
        <div style={{ marginBottom: '15px' }}>
          <button 
            onClick={() => setShowHeatmap(!showHeatmap)}
            style={{ marginRight: '10px', padding: '8px', cursor: 'pointer', fontWeight: 'bold' }}
          >
            {showHeatmap ? 'Hide Heatmap' : 'Show Heatmap'}
          </button>
          
          <button 
            onClick={() => setShowMarkers(!showMarkers)}
            style={{ padding: '8px', cursor: 'pointer', fontWeight: 'bold' }}
          >
            {showMarkers ? 'Hide Markers' : 'Show Markers'}
          </button>
        </div>

        {/* Map Rendering Boundaries View Canvas Box Container */}
        <div style={{ height: '600px', width: '100%', borderRadius: '8px', overflow: 'hidden' }}>
          <Map
            defaultZoom={13}
            defaultCenter={{ lat: 37.7749, lng: -122.4194 }}
            mapId="YOUR_MAP_ID_OPTIONAL" // Provide a custom map ID style layout theme if available
          >
            {/* Conditional Layer Generation Mount Hooks */}
            {showHeatmap && <HeatmapLayer points={heatmapData} />}

            {showMarkers && markerData.map(marker => (
              <Marker 
                key={marker.id}
                position={{ lat: marker.lat, lng: marker.lng }}
                title={marker.title}
                clickable={true}
                onClick={() => alert(`Clicked target point pin node: ${marker.title}`)}
              />
            ))}
          </Map>
        </div>

      </div>
    </APIProvider>
  );
}
